## Code
{% highlight cpp %}

#include <iostream>
#include <algorithm>
using namespace std;

int n;
int cost[10000];
int limit;
int value;

int main()
{
    int i;
    
    scanf("%d", &n);
    
    for (i = 0; i < n; i++) {
        scanf("%d", &cost[i]);
        if (cost[i] > value)
            value = cost[i];
    }
    scanf("%d", &limit);
    
    int l = 0, r = value, c;
    int sum;
    
    while (l <= r) {
        c = (l + r) >> 1;
        
        sum = 0;
        for (i = 0; i < n; i++) {
            if (cost[i] > c)
                sum += c;
            else
                sum += cost[i];
        }
        
        if (sum > limit)
            r = c - 1;
        else
            l = c + 1;
    }
    
    printf("%d\n", r);
    
    return 0;
}
